# chrome 
brew cask install google-chrome

# java 
brew cask install java # pathも通ってた

# eclipse
brew cask install eclipse-ide
/Applications/Eclipse.app/Contents/eclipse.iniの最後に
-Dfile.encoding=UTF-8

# php (xampp)
brew search xampp
Applications/XAMPP/xamppfiles/etc/php.iniにencodeing,charsetとtimezone,order,languageを変更

# neovim
brew install neovim/neovim/neovim

~/.bash_profileに
export XDG_CONFIG_HOME="$HOME/.config"
alias vim=nvim

# dein
#まず
$ mkdir ~/.cache/dein
#とした後に
$ cd ~/.cache/dein
#で移動してその状態で
$ curl https://raw.githubusercontent.com/Shougo/dein.vim/master/bin/installer.sh > installer.sh
$ sh ./installer.sh ~/.cache/dein
#と打ってあげる(~/.cache/deinではなく、~/.vim配下にした)
そして
~/.config/nvim/init.vimに吐き出されたdein Scriptを貼り付け
